Разработка интерактивного гида по японской мифологии и культуре

Выпускная квалификационная работа

Краснова Ангелина, группа 9ИСИП42-22(26)

Актуальность и проблема

Отсутствие доступных русскоязычных ресурсов, сочетающих энциклопедические знания о японской мифологии с современными интерактивными технологиями.

Существующие решения либо статичны, либо требуют установки приложений.

Цель

Разработать и реализовать интерактивный веб-гид по японской мифологии с интеграцией технологий дополненной реальности и адаптивным интерфейсом.

Задачи

1. Анализ японской мифологии и существующих аналогов

2. Проектирование архитектуры и дизайн-концепции

3. Разработка адаптивного интерфейса

4. Интеграция AR-функционала

5. Публикация и тестирование

  --text-light: #F0F0F0;
  --text-muted: #AAAAAA;
}
/* Загрузка данных из JSON */
async function loadCreaturesData() {
  const response = await fetch('data/creatures.json');
  const data = await response.json();
  const creatures = data.creatures;
  displayCatalog(creatures);
  initFilters(creatures);
}
/* Фильтрация каталога по типам */
function initFilters(creatures) {
  const filterBtns = document.querySelectorAll('.filter-btn');
  const cards = document.querySelectorAll('.preview-card');
  filterBtns.forEach(btn => {
    btn.addEventListener('click', function() {
      const filter = this.dataset.filter;
      cards.forEach(card => {
        if (filter === 'all' || card.dataset.type === filter) {
          card.style.display = 'block';
        } else {
          card.style.display = 'none';
        }
      });});});}
/* Динамическое создание карточек */
function displayCatalog(creatures) {
  const catalogGrid = document.getElementById('catalogGrid');
  catalogGrid.innerHTML = '';
  creatures.forEach(creature => {
    const card = document.createElement('div');
    card.className = 'preview-card';
    card.dataset.type = creature.type;
    card.innerHTML = `<h3>${creature.name}</h3>`;
    catalogGrid.appendChild(card);
  });
}
